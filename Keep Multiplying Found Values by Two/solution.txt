/**
 * @param {number[]} nums
 * @param {number} original
 * @return {number}
 */
var findFinalValue = function(nums, original) {
    const set = new Set();
    for(let num of nums){
        set.add(num);
    }
    while(true){
        if(set.has(original)){
            original *= 2;
        }else break;
    }
    return original;
};
