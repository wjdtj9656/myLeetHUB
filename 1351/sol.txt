/**
 * @param {number[][]} grid
 * @return {number}
 */
var countNegatives = function(grid) {
    let count = 0;
    const rows = grid.length;
    const cols = grid[0].length;
    let r = 0;
    let c = cols - 1;

    while (r < rows && c >= 0) {
        if (grid[r][c] < 0) {
            count += (rows - r);
            c--;
        } else {
            r++;
        }
    }

    return count;
};
