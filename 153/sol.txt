/**
 * @param {number[]} nums
 * @return {number}
 */
var findMin = function(nums) {
    let left = 0;
    let right = nums.length - 1;

    while (left < right) {
        let mid = Math.floor((left + right) / 2);

        if (nums[mid] > nums[right]) {
            // 최소값이 mid 오른쪽에 있는 경우
            left = mid + 1;
        } else {
            // nums[mid] <= nums[right] 인 경우, mid가 최소값일 수도 있으므로 포함하여 왼쪽 탐색
            right = mid;
        }
    }

    // left와 right가 만나는 지점이 최소값의 인덱스입니다.
    return nums[left];
};
